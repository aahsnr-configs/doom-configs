** Parenthesis
#+begin_src emacs-lisp
;;; ============================================================
;;; Org Source Block Bracket Syntax Fix
;;; ============================================================
;;; In org-mode, < and > are matching delimiters in the syntax table.
;;; This causes false mismatch errors in code like (> 2 4) inside
;;; org source blocks. This fix changes them to regular punctuation,
;;; but only outside strings and comments, and only inside src blocks.

;;;###autoload
(defun my/org-src-bracket-syntax-fix (start end)
  "Fix syntax of < and > inside org source blocks.
Changes < and > to regular punctuation, but only outside strings
and comments, and only inside src blocks."
  (when (eq major-mode 'org-mode)
    (save-excursion
      (goto-char start)
      (while (re-search-forward "[<>]" end t)
        (let ((match-pos (1- (point))))
          (save-excursion
            (when (and (re-search-backward
                        "[[:space:]]*#\\+\\(begin\\|end\\)_src\\_>" nil t)
                       (string-equal (downcase (match-string 1)) "begin"))
              (let ((ppss (syntax-ppss match-pos)))
                (unless (or (nth 3 ppss)   ; inside string
                            (nth 4 ppss))  ; inside comment
                  (put-text-property match-pos (1+ match-pos)
                                     'syntax-table (string-to-syntax "_")))))))))))

;; Store org's original syntax-propertize-function so we can call it
;; from our wrapper instead of replacing it entirely.
(defvar-local my/org-original-syntax-propertize-function nil
  "Stores the original `syntax-propertize-function' for org buffers.")

(defun my/org-syntax-propertize-wrapper (start end)
  "Call org's original syntax propertize function, then fix < > in src blocks."
  (when (functionp my/org-original-syntax-propertize-function)
    (funcall my/org-original-syntax-propertize-function start end))
  (my/org-src-bracket-syntax-fix start end))

(after! org
  (add-hook! 'org-mode-hook
    (defun my/org-src-syntax-fix-h ()
      "Wrap org's syntax-propertize-function to fix angle brackets in src blocks."
      (setq-local my/org-original-syntax-propertize-function
                  syntax-propertize-function)
      (setq-local syntax-propertize-function
                  #'my/org-syntax-propertize-wrapper))))

;;; ============================================================
;;; Replace Smartparens with Built-in electric-pair-mode
;;; ============================================================

;; 1. Prevent smartparens from being activated globally.
;;    Per Doom's FAQ, this is the correct way to disable smartparens.
;;    We cannot use (package! smartparens :disable t) because it's
;;    a core package and will break Doom's base config.
(remove-hook 'doom-first-buffer-hook #'smartparens-global-mode)

;; 2. Remove Doom's hook that disables smartparens navigation.
;;    This hook is added when +smartparens flag is enabled in init.el.
;;    It's harmless when smartparens is off, but we remove it for cleanliness.
(after! smartparens
  (remove-hook! 'after-change-major-mode-hook
    #'doom-disable-smartparens-navigate-skip-match-h))

;; 3. Enable electric-pair-mode globally.
;;    This is Emacs' built-in delimiter pairing system.
;;    It provides:
;;      - Auto-pairing of (), [], {}, "", '', etc. based on syntax tables
;;      - Region wrapping (select text, type ( to wrap it)
;;      - Skip-over closing delimiters (type ) when already at ) to move past)
;;      - Balance preservation (won't create unbalanced delimiters)
;;      - Adjacent pair deletion (backspace between () deletes both)
(electric-pair-mode 1)

;; 4. Configure electric-pair-mode settings.
(setq-default
 ;; Skip whitespace when deciding whether to skip over closing delimiter
 electric-pair-skip-whitespace t
 ;; Open an extra newline when pressing RET between adjacent pairs
 electric-pair-open-newline-between-pairs t
 ;; Delete adjacent pairs when backspacing between them
 electric-pair-delete-adjacent-pairs t)

;; 5. In org-mode, inhibit < pairing to not interfere with org-tempo.
;;    This allows org-tempo templates like <s to work without electric-pair
;;    inserting a matching >.
(after! org
  (add-hook! 'org-mode-hook
    (defun my/org-disable-angle-bracket-pairing-h ()
      "Inhibit < pairing in org-mode for org-tempo compatibility."
      (setq-local electric-pair-inhibit-predicate
                  (lambda (c)
                    (or (funcall (default-value 'electric-pair-inhibit-predicate) c)
                        (char-equal c ?<)))))))

;; 6. Enable echo area display for offscreen matching brackets.
;;    Doom already enables show-paren-mode for bracket highlighting.
;;    This variable (added in Emacs 29.1) shows the matching bracket's
;;    line in the echo area when the matching bracket is offscreen.
;;    Options: nil (off), t (echo area), 'overlay, 'child-frame
(when (boundp 'show-paren-context-when-offscreen)
  (setq show-paren-context-when-offscreen t))
#+end_src

** Transient Template Menu
#+begin_src emacs-lisp
(after! (org transient)
  ;; Helper function to expand org templates
  (defun ar/org-template-expand (str &optional mod)
    "Expand org template STR with optional modifier MOD."
    (let (text)
      (when (region-active-p)
        (setq text (buffer-substring (region-beginning) (region-end)))
        (delete-region (region-beginning) (region-end)))
      (insert str)
      (org-tempo-complete-tag)
      (when mod (insert mod) (forward-line))
      (when text (insert text))))

  ;; Specific template actions
  (defun ar/org-insert-perl-tangled ()
    (interactive)
    (insert "#+HEADERS: :results output :exports both :shebang \"#!/usr/bin/env perl\"\n")
    (ar/org-template-expand "<s" "perl"))

  (defun ar/org-insert-ipython ()
    (interactive)
    (ar/org-template-expand "<s" "ipython :session :exports both :results raw drawer\n$0"))

  ;; Title generator
  (defun ar/org-transient-title ()
    (concat
     (if (and (fboundp 'nerd-icons-sucicon) (display-graphic-p))
         (nerd-icons-sucicon "nf-custom-orgmode" :face 'transient-heading :v-adjust 0.02)
       "")
     (propertize " Org Template" 'face 'transient-heading)))

  ;; The transient menu definition
  (transient-define-prefix ar/org-template-transient ()
    "Org Mode Structure Templates"
    ["" :description ar/org-transient-title

     ["Basic"

* LaTeX
** Engine and Compilation Setup
Configure latexmk as the build system with LuaLaTeX as the modern default engine. Doom's latex module (+cdlatex +fold) already provides AUCTeX, CDLaTeX, folding support, and SyncTeX integration. We extend it with latexmk for robust multi-pass compilation.

#+begin_src emacs-lisp
(after! tex
  (setq TeX-engine 'luatex)
  (setq TeX-output-dir "build/")
  (setq LaTeX-command "latex -shell-escape")
  (setq TeX-save-query nil)
  (setq TeX-clean-confirm nil)
  (setq font-latex-fontify-script 'multi-level
        font-latex-fontify-sectioning 1.2))

(after! latex
  (add-to-list 'TeX-command-list
               '("LuaLaTeXmk" "latexmk -pdflua -pdflualatex='lualatex -shell-escape -interaction=nonstopmode' -f -outdir=%o %S"
                 TeX-run-TeX nil t
                 :help "Run latexmk with LuaLaTeX") t)

  ;; Add continuous preview mode (watch mode)
  (add-to-list 'TeX-command-list
               '("LaTeXmk-Watch" "latexmk -pvc -pdflua -pdflualatex='lualatex -shell-escape -interaction=nonstopmode' -outdir=%o %S"
                 TeX-run-TeX nil t
                 :help "Run latexmk in continuous preview mode") t)

  (setq-default TeX-command-default
                (cond ((eq TeX-engine 'luatex) "LuaLaTeXmk")
                      ((eq TeX-engine 'xetex) "XeLaTeXmk")
                      (t "LaTeXmk")))

  ;; Auxiliary files to clean
  (setq LaTeX-clean-intermediate-suffixes
        '("\\.aux" "\\.bbl" "\\.blg" "\\.brf" "\\.fot"
          "\\.glo" "\\.gls" "\\.idx" "\\.ilg" "\\.ind"
          "\\.lof" "\\.log" "\\.lot" "\\.nav" "\\.out"
          "\\.snm" "\\.toc" "\\.url" "\\.synctex\\.gz"
          "\\.fdb_latexmk" "\\.fls" "\\.xdv"
          "-blx\\.bib" "\\.run\\.xml"))

  )

;; Configure pdf-tools integration with AUCTeX
(setq +latex-viewers '(pdf-tools))
(after! tex
  (setq TeX-view-program-selection '((output-pdf "PDF Tools"))
        TeX-view-program-list '(("PDF Tools" TeX-pdf-tools-sync-view)))

  ;; Automatically revert PDF buffer after compilation
  (add-hook 'TeX-after-compilation-finished-functions
            #'TeX-revert-document-buffer))
